    async function processData() {
        const subFile = document.getElementById('fileSub').files[0];
        const vouFile = document.getElementById('fileVou').files[0];

        if (!subFile || !vouFile) {
            log("错误：请确保两个文件都已上传！", "error");
            return;
        }

        try {
            log("开始解析源文件...", "info");

            // 1. 读取源文件（使用智能表头定位）
            const subData = await readExcelSmart(subFile, ['科目编码', '编码', '科目名称']);
            const vouData = await readExcelSmart(vouFile, ['凭证号', '凭证日期', '科目']);

            if (!subData.length || !vouData.length) throw new Error("文件读取失败或内容为空");

            const voucherResult = convertVouchers(vouData, auxItemSet);

            // 4. 处理科目
            log("正在生成科目导入数据...", "info");
            const subjectResult = convertSubjects(subData);

            // 5. 处理辅助资料 (从凭证中提取的)
            log("正在生成辅助核算资料...", "info");
            const auxResult = Array.from(auxItemSet.values());

            // 6.1 导出凭证

    // --- 转换逻辑：凭证 (保持不变) ---
    function convertVouchers(data, auxItemSet) {
        let result = [];
        let lastDate = "";
        let lastWord = "记";
        let lastNo = "";
        let entrySeq = 0; // 分录序号

        data.forEach(row => {
            // 跳过无效行
            if (!row['科目'] && !row['借方金额'] && !row['贷方金额'] && !row['分录摘要']) return;
            if (String(row['凭证号']).includes('合计')) return;

            // 1. 填充日期和凭证号 (Fill-down)
            if (row['凭证号']) {
                lastDate = row['凭证日期'];
                let vStr = String(row['凭证号']);
                if (vStr.includes('-')) {
                    lastWord = vStr.split('-')[0];
                    lastNo = vStr.split('-')[1];
                } else {
                    lastNo = vStr;
                }
                entrySeq = 0; // 新凭证重置序号
            }
            entrySeq++;

            // 2. 拆解科目
            let subCode = "";
            let subName = "";
            if (row['科目']) {
                const parts = row['科目'].split(" ");
                subCode = parts[0];
                subName = parts.length > 1 ? parts.slice(1).join(" ") : "";
            }

            // 3. 构建金蝶行结构
            let kRow = {
                "日期": lastDate,
                "凭证字": lastWord,
                "凭证号": lastNo,
                "附件数": "",
                "分录序号": entrySeq,
                "摘要": row['分录摘要'] || row['摘要'] || "",
                "科目代码": subCode,
                "科目名称": subName,
                "借方金额": row['借方金额'] || 0,
                "贷方金额": row['贷方金额'] || 0,
                "客户": "", "供应商": "", "职员": "", "项目": "", "部门": "", "存货": "",
                "自定义辅助核算类别": "", "自定义辅助核算编码": "",
                "自定义辅助核算类别1": "", "自定义辅助核算编码1": "",
                "数量": row['数量'] || "",
                "单价": row['单价'] || "",
                "原币金额": row['借方原币'] || row['贷方原币'] || "",
                "币别": "RMB", // 默认为RMB
                "汇率": row['汇率'] || ""
            };

            // 4. 辅助核算解析与映射
            if (row['辅助核算']) {
                const matches = row['辅助核算'].matchAll(/【\s*(.*?)\s*:\s*(.*?)\s*\/\s*(.*?)\s*】/g);
                let customItems = []; 

                for (const match of matches) {
                    let type = match[1].trim();
                    let code = match[2].trim();
                    let name = match[3].trim();

                    if (type === '人员') type = '职员';

                    auxItemSet.set(`${type}_${code}`, { "类别": type, "编码": code, "名称": name });

                    if (["客户", "供应商", "职员", "项目", "部门", "存货"].includes(type)) {
                        kRow[type] = code; 
                    } else {
                        customItems.push({ type, code }); 
                    }
                }

                if (customItems.length > 0) {
                    kRow["自定义辅助核算类别"] = customItems[0].type;
                    kRow["自定义辅助核算编码"] = customItems[0].code;
                }
                if (customItems.length > 1) {
                    kRow["自定义辅助核算类别1"] = customItems[1].type;
                    kRow["自定义辅助核算编码1"] = customItems[1].code;
                }
            }

            result.push(kRow);
        });
        return result;
    }

    // --- 新增工具：根据图片规则判断金蝶类别 ---
    function getKingdeeCategory(code) {
        // 提取前4位作为一级科目代码进行判断
        const topCode = parseInt(code.substring(0, 4));
        if (isNaN(topCode)) return "其他";

        // 1. 资产类
        if (topCode >= 1001 && topCode <= 1499) return "流动资产";
        if (topCode >= 1501 && topCode <= 1999) return "非流动资产";
        
        // 2. 负债类
        if (topCode >= 2001 && topCode <= 2499) return "流动负债";
        if (topCode >= 2501 && topCode <= 2999) return "非流动负债";
        
        // 3. 权益类
        if (topCode >= 4000 && topCode <= 4999) return "所有者权益";
        
        // 4. 损益类 - 细分规则
        if (topCode === 6001) return "营业收入";
        if (topCode >= 6051 && topCode <= 6301) return "其他收益"; // 含其他业务收入、汇兑、投资收益、营业外收入
        if (topCode >= 6401 && topCode <= 6403) return "营业成本及税金"; // 含主营成本、其他成本、税金及附加
        if (topCode >= 6601 && topCode <= 6604) return "期间费用"; // 含销售、管理、财务
        if (topCode >= 6701 && topCode <= 6711) return "其他损失"; // 含资产减值、营业外支出
        if (topCode === 6801) return "所得税";
        if (topCode === 6901) return "以前年度损益调整";
        
        // 5. 兜底逻辑 (处理成本类和共同类)
        const firstChar = code.charAt(0);
        if (firstChar === '5') return "成本"; // 生产成本等
        if (firstChar === '3') return "共同";
        
        return "其他";
    }

    // --- 转换逻辑：科目 (已调整) ---
    function convertSubjects(data) {
        return data.map(row => {
            const code = String(row['编码'] || row['科目编码'] || "");
            const name = String(row['名称'] || row['科目名称'] || "");
            if (!code) return null;

            // 1. 类别判断 (调用新函数)
            const category = getKingdeeCategory(code);

            // 2. 余额方向判断
            let direction = "借";
            const firstChar = code.charAt(0);
            if (row['余额方向']) {
                direction = row['余额方向'];
            } else {
                // 默认推断
                if (['2', '3'].includes(firstChar) || (firstChar === '5' && code.startsWith('50'))) {
                    direction = "贷";
                }
            }

            // 3. 是否现金科目 (根据名称)
            let isCash = "否";
            if (name.includes("现金") || name.includes("银行") || code.startsWith("1001") || code.startsWith("1002")) {
                isCash = "是";
            }

            // 4. 辅助核算格式化
            let auxCat = "";
            let rawAux = row['辅助核算'];
            
            if (rawAux && rawAux.trim() !== "") {
                // 用友格式 "部门*|人员" 转为 金蝶格式 "部门/职员"
                auxCat = rawAux.replace(/人员/g, "职员").replace(/\*\|/g, "/").replace(/\*/g, "").replace(/\|/g, "/");
            }

            // 严格对应的字段名
            return {
                "编码": code,
                "名称": name,
                "类别": category,
                "余额方向": direction,
                "是否现金科目": isCash,
                "辅助核算类别": auxCat
            };
        }).filter(item => item !== null);
    }

    // --- 工具：智能读取Excel (保持不变) ---
    function readExcelSmart(file, keywords) {
        return new Promise((resolve, reject) => {
            const reader = new FileReader();
            reader.onload = (e) => {
                try {
                    const data = new Uint8Array(e.target.result);
                    const workbook = XLSX.read(data, { type: 'array' });
                    const sheet = workbook.Sheets[workbook.SheetNames[0]];
                    
                    const rawRows = XLSX.utils.sheet_to_json(sheet, { header: 1 });
                    let headerIndex = 0;
                    for (let i = 0; i < Math.min(rawRows.length, 15); i++) {
                        const rowStr = JSON.stringify(rawRows[i]);
                        if (keywords.some(k => rowStr.includes(k))) {
                            headerIndex = i;
                            break;
                        }
                    }
                    
                    const jsonData = XLSX.utils.sheet_to_json(sheet, { range: headerIndex, defval: "" });
                    resolve(jsonData);
                } catch (err) { reject(err); }
            };
            reader.readAsArrayBuffer(file);
        });
    }

    // --- 工具：导出Excel (保持不变) ---
    function exportToExcel(jsonData, headers, fileName) {
        const ws = XLSX.utils.json_to_sheet(jsonData, { header: headers });
        const wb = XLSX.utils.book_new();
        XLSX.utils.book_append_sheet(wb, ws, "Sheet1");
        XLSX.writeFile(wb, fileName);
    }
